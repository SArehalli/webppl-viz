var model = function() {
  var x = uniform(0,1);
  var y = uniform(0,1);

  factor(Math.log(Math.abs(x-y)))
  return {x: x, y: y};
}

viz.vegaPrint(MH(model, 5000))
var model = MH(function() {
  var brand = flip(0.7) ? 'coke' : 'pepsi';
  return {
    brand: brand,
    price: gaussian(brand == 'coke' ? 4 : 2, 1)
  }
}, 2000);
viz.vegaPrint(model)
